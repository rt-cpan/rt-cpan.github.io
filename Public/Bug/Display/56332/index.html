'?openid-check=1', @@ -122,8 +122,7 @@ my $user = +{ map { $_ => scalar $identity->$_ } qw( url display rss atom foaf declared_rss declared_atom declared_foaf foafmaker ) }; # Dude, I did not design the array as hash spec.
