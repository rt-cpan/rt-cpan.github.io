5.421 has return IO::File->new(\($self->{MBS_Data}), $mode); where 5.420 has return new IO::Scalar \($self->{MBS_Data}); Now, my $b = MIME::Body::InCore->new(\"hi"); my $fh = $b->open("r"); print <$fh>; prints SCALAR(0x....)
