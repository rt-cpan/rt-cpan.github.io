package My::Module; use Moose; use MooseX::Types::LoadableClass qw{LoadableClass}; has thing => ( is => 'rw', coerce => 1, required => 1, lazy => 1, isa => LoadableClass, default => 'Module::I::Dont::Have', ); 1; package main; my $obj = My::Module->new; print $obj->thing;
