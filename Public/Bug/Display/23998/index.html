Test case: use Class::Date qw(date); my $dt = date("2006-06-24 05:23:42", "CET"); print($dt->to_tz("GMT")->strftime("%Y-%m-%d %H:%M:%S%z") . "\n" . $dt->strftime("%Y-%m-%d %H:%M:%S%z") .
