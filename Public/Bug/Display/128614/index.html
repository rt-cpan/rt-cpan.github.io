#!/usr/bin/env perl use warnings; use strict; use Chart::Gnuplot; my @x = (-10 .. 10); my @y = (0 .. 20); my $chart = Chart::Gnuplot->new( output => "~/plot.png", title => "Plot example", xlabel => "x", ylabel => "y" ); my $dataset = Chart::Gnuplot::DataSet->new( xdata => \@x, ydata => \@y, style => "points", ); my $func = Chart::Gnuplot::DataSet->new( func => "3*x+4" ); $chart->plot2d($dataset, $func);
