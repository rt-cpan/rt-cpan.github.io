  my $imdbObj = undef;

    $imdbObj = new IMDB::Film(crit => $imdb2);
  };

  if ($@) {
    print "IMDBinfo::GetInfo($uid,$imdb) failed to get URL for imdb $imdb2\n";
    print "IMDBinfo::GetInfo() $@\n";
  } else {
    print "IMDBinfo::GetInfo() for $imdb2\n";
  }
