Subject: High memory usage
The following script use strict; use warnings; use HTTP::Daemon; use HTTP::Daemon::SSL; my $daemon = HTTP::Daemon::SSL->new(LocalAddr => '', LocalPort => 9903, ReuseAddr => 1); my $conn = $daemon->accept; my $request = $conn->get_request(); die unless $request; my $resp = HTTP::Response->new(200, "OK"); my $buf = 'x' x 2_000_000_000; $resp->content($buf); $conn->force_last_request(); $conn->send_response($resp); __END__ eats 13Gb of resident memory (on 16Gb system) when serves the request.
