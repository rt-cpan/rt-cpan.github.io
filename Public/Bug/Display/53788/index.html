#!/usr/bin/perl use strict; use warnings; package XSTest; use base 'Template::Stash::XS'; sub new { my $class = shift; my $self = Template::Stash::XS->new( @_ ); return bless $self, $class; } sub undefined { return undef; } #---------------------------------------------------------------------- package nonXSTest; use base 'Template::Stash'; sub new { my $class = shift; my $self = Template::Stash->new( @_ ); return bless $self, $class; } sub undefined { return undef; } #---------------------------------------------------------------------- package main; my $stash = nonXSTest->new(); my $xs_stash = XSTest->new(); my $template_text = q{[% SET my_undef = nothing; my_undef.defined ? 'fail' : 'succeed'; %] }; print 'non-XS stash ' . ( defined $stash->undefined() ? 'fail' : 'succeed' ); print "\n"; print 'XS stash ' . ( defined $xs_stash->undefined() ? 'fail' : 'succeed' ); print "\n"; print 'non-XS template '; Template->new( { STASH => $stash, })->process( \$template_text ); print 'XS template '; Template->new( { STASH => $xs_stash, })->process( \$template_text );
